## Tags Example

Now, Here we will discuss about the tags. Below is Example of the tags:

```bash
index.html

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>HTML Tutorials</title>
</head>
<body>
    <section>
        <!-- Anchor tag -->
        <a href="https://example.com">This is a hyperlink</a>

        <!-- Article tag -->
        <article>
            <h2>Article Title</h2>
            <p>This is an article.</p>
        </article>

        <!-- Aside tag -->
        <aside>
            <p>This is an aside.</p>
        </aside>

        <!-- Audio tag -->
        <audio controls>
            <source src="audio-file.mp3" type="audio/mpeg">
            Your browser does not support the audio element.
        </audio>

        <!-- Bold tag -->
        <b>This text is bold</b>

        <!-- Blockquote tag -->
        <blockquote>
            This is a blockquote from another source.
        </blockquote>

        <!-- Line break tag -->
        <br>

        <!-- Button tag -->
        <button>Click Me</button>

        <!-- Canvas tag -->
        <canvas id="myCanvas"></canvas>

        <!-- Caption tag -->
        <table>
            <caption>This is a table caption</caption>
            <tr><td>Data</td></tr>
        </table>

        <!-- Code tag -->
        <code>console.log('Hello, world!');</code>

        <!-- Description details tag -->
        <dl>
            <dt>Term</dt>
            <dd>Definition of the term</dd>
        </dl>

        <!-- Division tag -->
        <div>This is a div</div>

        <!-- Description list tag -->
        <!-- See above for dl example -->

        <!-- Description term tag -->
        <!-- See above for dt example -->

        <!-- Emphasized text tag -->
        <em>This text is emphasized</em>

        <!-- Fieldset tag -->
        <fieldset>
            <legend>Personal Information</legend>
            <label for="name">Name:</label>
            <input type="text" id="name" name="name">
        </fieldset>

        <!-- Footer tag -->
        <footer>This is the footer</footer>

        <!-- Form tag -->
        <form>
            <label for="email">Email:</label>
            <input type="email" id="email" name="email">
            <input type="submit" value="Submit">
        </form>

        <!-- Header tag -->
        <header>
            <h1>This is the header</h1>
        </header>

        <!-- Horizontal rule tag -->
        <hr>

        <!-- iFrame tag -->
        <iframe src="https://example.com" width="300" height="200"></iframe>

        <!-- Image tag -->
        <img src="image.jpg" alt="Description of the image">

        <!-- Input tag -->
        <!-- See form example above -->

        <!-- Label tag -->
        <!-- See form example above -->

        <!-- Legend tag -->
        <!-- See fieldset example above -->

        <!-- List item tag -->
        <ul>
            <li>List item 1</li>
            <li>List item 2</li>
        </ul>

        <!-- Main tag -->
        <main>
            <p>This is the main content</p>
        </main>

        <!-- Mark tag -->
        <mark>This text is highlighted</mark>

        <!-- Marquee tag (deprecated) -->
        <marquee>This text scrolls</marquee>

        <!-- Navigation tag -->
        <nav>
            <a href="#home">Home</a>
            <a href="#about">About</a>
        </nav>

        <!-- Ordered list tag -->
        <ol>
            <li>Ordered item 1</li>
            <li>Ordered item 2</li>
        </ol>

        <!-- Option tag -->
        <select>
            <option value="1">Option 1</option>
            <option value="2">Option 2</option>
        </select>

        <!-- Paragraph tag -->
        <p>This is a paragraph</p>

        <!-- Preformatted text tag -->
        <pre>
            Preformatted text
            Line 2
        </pre>

        <!-- Section tag -->
        <section>
            <h2>Section Heading</h2>
            <p>This is a section</p>
        </section>

        <!-- Select tag -->
        <!-- See option example above -->

        <!-- Source tag -->
        <!-- See audio example above -->

        <!-- Span tag -->
        <span>This is a span</span>

        <!-- Strong tag -->
        <strong>This text is important</strong>

        <!-- SVG tag -->
        <svg width="100" height="100">
            <circle cx="50" cy="50" r="40" stroke="black" stroke-width="3" fill="red" />
        </svg>

        <!-- Table tag -->
        <!-- See caption example above -->

        <!-- Table body tag -->
        <table>
            <tbody>
                <tr>
                    <td>Body data</td>
                </tr>
            </tbody>
        </table>

        <!-- Table cell tag -->
        <!-- See above for td example -->

        <!-- Textarea tag -->
        <textarea rows="4" cols="50">This is a textarea</textarea>

        <!-- Table footer tag -->
        <table>
            <tfoot>
                <tr>
                    <td>Footer data</td>
                </tr>
            </tfoot>
        </table>

        <!-- Table header cell tag -->
        <table>
            <thead>
                <tr>
                    <th>Header cell</th>
                </tr>
            </thead>
        </table>

        <!-- Table row tag -->
        <table>
            <tr>
                <td>Row data</td>
            </tr>
        </table>

        <!-- Underline tag -->
        <u>This text is underlined</u>

        <!-- Unordered list tag -->
        <!-- See ul example above -->

        <!-- Video tag -->

            <source src="video-file.mp4" type="video/mp4">

    </section>
```
